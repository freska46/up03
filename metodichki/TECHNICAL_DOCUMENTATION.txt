1. ВЫБОР ТЕХНОЛОГИЙ
--------------------
- Язык: C# (.NET 8.0)
- Web API: ASP.NET Core 8 (минимальный шаблон)
- База данных: PostgreSQL + Entity Framework Core
- Драйвер: Npgsql.EntityFrameworkCore.PostgreSQL
- Настольное приложение: WPF (Windows Presentation Foundation)

Почему именно эти технологии:
- ASP.NET Core - самый популярный фреймворк для Web API
- WPF - встроен в .NET, не требует дополнительных установок
- PostgreSQL - бесплатная и мощная СУБД
- Entity Framework Core - ORM для работы с БД через код

3. КОД API (ASP.NET Core)
----------------------
Точка входа - Program.cs:
```csharp
using Microsoft.EntityFrameworkCore;
using PakDzal_Games_API.Data;

var builder = WebApplication.CreateBuilder(args);

// Строка подключения к БД
var connectionString = builder.Configuration.GetConnectionString("DefaultConnection") 
    ?? "Host=localhost;Database=pakdzal_games;Username=postgres;Password=111";

// Настройка Entity Framework
builder.Services.AddDbContext<GameClubDbContext>(options =>
    options.UseNpgsql(connectionString));

// Добавляем контроллеры
builder.Services.AddControllers();

// Swagger для документации
builder.Services.AddSwaggerGen();

var app = builder.Build();

app.UseSwagger();
app.UseSwaggerUI();

app.MapControllers();
app.Run();
```

Контроллер (API эндпоинты):
```csharp
using Microsoft.AspNetCore.Mvc;
using Microsoft.EntityFrameworkCore;
using PakDzal_Games_API.Data;
using PakDzal_Games_API.Models;

[ApiController]
[Route("api/[controller]")]
public class UsersController : ControllerBase
{
    private readonly GameClubDbContext _context;

    public UsersController(GameClubDbContext context)
    {
        _context = context;
    }

    // GET /api/users
    [HttpGet]
    public async Task<ActionResult<IEnumerable<User>>> GetUsers()
    {
        return await _context.Users.ToListAsync();
    }

    // GET /api/users/{id}
    [HttpGet("{id}")]
    public async Task<ActionResult<User>> GetUser(int id)
    {
        var user = await _context.Users.FindAsync(id);
        if (user == null) return NotFound();
        return user;
    }

    // POST /api/users
    [HttpPost]
    public async Task<ActionResult<User>> CreateUser(User user)
    {
        _context.Users.Add(user);
        await _context.SaveChangesAsync();
        return CreatedAtAction(nameof(GetUser), new { id = user.UserId }, user);
    }

    // PUT /api/users/{id}
    [HttpPut("{id}")]
    public async Task<IActionResult> UpdateUser(int id, User user)
    {
        if (id != user.UserId) return BadRequest();
        _context.Entry(user).State = EntityState.Modified;
        await _context.SaveChangesAsync();
        return NoContent();
    }

    // DELETE /api/users/{id}
    [HttpDelete("{id}")]
    public async Task<IActionResult> DeleteUser(int id)
    {
        var user = await _context.Users.FindAsync(id);
        if (user == null) return NotFound();
        _context.Users.Remove(user);
        await _context.SaveChangesAsync();
        return NoContent();
    }
}
```


4. КОД WPF ПРИЛОЖЕНИЯ
--------------------
ApiService (работа с HTTP):
```csharp
using System.Net.Http;
using System.Net.Http.Json;
using PakDzal_Games_WPF.Models;

public class ApiService
{
    private readonly HttpClient _httpClient;
    private const string BaseUrl = "http://localhost:5222/api";

    public ApiService()
    {
        _httpClient = new HttpClient();
    }

    public async Task<List<User>> GetUsersAsync(string? search = null, string? city = null)
    {
        var url = $"{BaseUrl}/users";
        if (!string.IsNullOrEmpty(search))
            url += $"?search={Uri.EscapeDataString(search)}";
        if (!string.IsNullOrEmpty(city))
            url += city.Contains("?") ? $"&city={Uri.EscapeDataString(city)}" 
                                   : $"?city={Uri.EscapeDataString(city)}";
        return await _httpClient.GetFromJsonAsync<List<User>>(url) ?? new List<User>();
    }

    public async Task<User> CreateUserAsync(User user)
    {
        var response = await _httpClient.PostAsJsonAsync($"{BaseUrl}/users", user);
        return (await response.Content.ReadFromJsonAsync<User>())!;
    }
    // ... остальные методы
}
```

10. СОЗДАНИЕ БД (pgAdmin)
------------------------
1. Открыть pgAdmin 4
2. Создать базу данных "pakdzal_games"
3. Выполнить SQL:
```sql
CREATE TABLE Users (
    user_id SERIAL PRIMARY KEY,
    name VARCHAR(100) NOT NULL,
    email VARCHAR(100) UNIQUE NOT NULL,
    phone VARCHAR(20),
    city VARCHAR(50) DEFAULT 'Курск',
    registration_date DATE DEFAULT CURRENT_DATE
);

CREATE TABLE Games (
    game_id SERIAL PRIMARY KEY,
    title VARCHAR(100) NOT NULL,
    genre VARCHAR(50) NOT NULL,
    price_per_hour DECIMAL(10,2) NOT NULL,
    available BOOLEAN DEFAULT TRUE
);

CREATE TABLE Sessions (
    session_id SERIAL PRIMARY KEY,
    user_id INTEGER REFERENCES Users(user_id),
    game_id INTEGER REFERENCES Games(game_id),
    start_time TIMESTAMP NOT NULL,
    end_time TIMESTAMP,
    total_price DECIMAL(10,2)
);

-- Тестовые данные
```
